const Sequelize = require('sequelize')
const db = require('../connection')

const X = db.define('X', {
    id: {
        type: Sequelize.INTEGER,
        primaryKey: true,
        autoIncrement: true
    },
    string: {
        type: Sequelize.STRING
    },
    number: {
        type: Sequelize.INTEGER
    }
}, {tableName: 'X'})

module.exports = X
